     curl -X POST http://localhost:3001/api/complete \

       }'

🎯 TO USE IN YOUR REACT APP
════════════════════════════════════════════════════════════════════════════

const API_URL = 'http://localhost:3001';

// Start assessment
const res = await fetch(`${API_URL}/api/start`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    name: 'John Doe',
    email: 'john@example.com',
    companyName: 'Acme Corp',
    stage: 'seed',
    vertical: 'saas-b2b',
    usesAi: false,
    physicalProduct: false
  })
});
const { sessionId } = await res.json();

// Submit responses
await fetch(`${API_URL}/api/respond`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    sessionId: sessionId,
    domain: 'technical',
    questionIndex: 0,
    value: 4
  })
});

// Complete assessment
const completeRes = await fetch(`${API_URL}/api/complete`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ sessionId: sessionId })
});
const result = await completeRes.json();
console.log('Assessment saved!', result.assessmentId);

// Get past assessments
const historyRes = await fetch(`${API_URL}/api/user/john@example.com/assessments`);
const history = await historyRes.json();
console.log('Past assessments:', history.assessments);
